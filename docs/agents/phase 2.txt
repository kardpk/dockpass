'use client'


Manages all wizard state.
Renders progress bar + current step.
Handles forward/back navigation.
Applies step transitions via Framer Motion.


PROGRESS BAR:
  Full-width bar at top (not inside card)
  Background: #D0E2F3 (light blue)
  Fill: #0C447C (navy)
  Height: 4px
  Transition: width 300ms ease-in-out
  
  Percentage: (currentStep - 1) / 6 * 100
  Step 1 = 0%, Step 2 = 16.6%, ... Step 6 = 83.3%
  On complete = 100%


STEP HEADER:
  Below progress bar, padding 20px 16px 0
  Small label: "Step [n] of 6" (11px, grey)
  Step title (18px, 600, navy):
    Step 1: "Your boat"
    Step 2: "Marina & dock"
    Step 3: "Captain profile"
    Step 4: "Trip defaults"
    Step 5: "Liability waiver"
    Step 6: "Add-on menu"


BACK BUTTON:
  Top-left, visible from step 2 onward
  ← (ChevronLeft icon) + "Back"
  Ghost style, 13px
  Triggers: setDirection(-1), setStep(step - 1)


STEP CONTENT AREA:
  Padding: 24px 16px
  Max-width: 520px, centred on desktop
  Mobile: full width


STEP TRANSITIONS (Framer Motion):
  import { AnimatePresence, motion } from
    'framer-motion'
  
  Wrap each step in motion.div
  Key: currentStep (triggers re-animation)
  
  Variants:
    enter: x = direction > 0 ? '100%' : '-100%'
    center: x = 0
    exit: x = direction < 0 ? '100%' : '-100%'
  
  Transition: duration 0.25s, ease 'easeInOut'
  AnimatePresence mode="wait"


STATE MANAGEMENT:
  const [step, setStep] = useState(1)
  const [direction, setDirection] = useState(1)
  const [data, setData] = useState(INITIAL_WIZARD_DATA)
  const [isSubmitting, setIsSubmitting] = useState(false)


  function goNext(newData: Partial<WizardData>) {
    setData(prev => ({ ...prev, ...newData }))
    setDirection(1)
    setStep(prev => prev + 1)
    // Scroll to top on step change
    window.scrollTo({ top: 0, behavior: 'smooth' })
  }


  function goBack() {
    setDirection(-1)
    setStep(prev => prev - 1)
    window.scrollTo({ top: 0, behavior: 'smooth' })
  }


Each step component receives:
  - data: WizardData (current state)
  - onNext: (partial: Partial<WizardData>) => void
  - isSubmitting: boolean (step 6 only)
